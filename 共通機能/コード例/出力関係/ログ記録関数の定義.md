```typescript
import { getHostObject } from "@/plugin";

/**  部局システムログ記録関数 */
const registTblBukyokulog = async (jobtype: string, bikou: string) => {
  // クライアント情報取得
  const clientInfo = JSON.parse(
    (await getHostObject()?.JavaScriptInterface.GetClientInformation()) ??
      "null",
  );

  // ログデータ構築
  const logInput = <TblBukyokulogInput>{
    datetime: new Date(),
    jobtype: jobtype,
    username: session.userCode,
    compname: clientInfo?.ComputerName,
    compuser: clientInfo?.UserName,
    bikou: bikou,
  };

  // ログ登録API呼び出し
  await services.api.registTblBukyokulog({ input: logInput });
};
```
