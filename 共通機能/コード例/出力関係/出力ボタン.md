/** [必須] 出力ボタン押下イベント */
const OnClickFileOutputButton = async () => {
  // [必須] データ件数チェック
  if (tableItems.value.length <= 0) {
    ElNotification({
      type: "warning",
      title: t("csvOutput.notification.noData.title"),
    });
    return;
  }

  // [必須] ダイアログを開いて出力項目選択
  const res =
    await alDataOutputDefinitionDialogRef.value!.open(outputDefinitionMap);
  if (!res) return; // キャンセル時は処理中断

  // [必須] ローディング表示
  const loading = alLoadingRef.value!.startLoading(() => {});
  try {
    // [必須] ファイル出力実行
    await res.write(tableItems.value);
  } finally {
    loading.close();
  }

  // [必須] 成功メッセージ表示
  ElNotification({
    type: "success",
    title: t("csvOutput.notification.success.title"),
  });
};
